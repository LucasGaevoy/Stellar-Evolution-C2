    function getSmoothPoint(data, pct) {
        if (pct <= 0) return data[0];
        if (pct >= 1) return data[data.length - 1];
        const n = data.length - 1;
        const pos = pct * n;
        const i = Math.floor(pos);
        const t = pos - i;
        const p0 = data[Math.max(0, i - 1)];
        const p1 = data[i];
        const p2 = data[Math.min(n, i + 1)];
        const p3 = data[Math.min(n, i + 2)];
        const crm = (v0, v1, v2, v3, t) => {
            return 0.5 * ((2 * v1) + (-v0 + v2) * t + (2 * v0 - 5 * v1 + 4 * v2 - v3) * t * t + (-v0 + 3 * v1 - 3 * v2 + v3) * t * t * t);
        };
        return {
            T: crm(p0.T, p1.T, p2.T, p3.T, t),
            L: Math.pow(10, crm(Math.log10(p0.L), Math.log10(p1.L), Math.log10(p2.L), Math.log10(p3.L), t))
        };

      function Labels(text, T, Mag, angle=0){
        ctx.save();
        ctx.translate(TempConversion(T), YAxis(Mag)); ctx.rotate(angle);
        ctx.font = "12px sans-serif"; ctx.fillStyle = "rgba(0,0,0,0.7)";
        ctx.textAlign = "center"; ctx.fillText(text,0,0);
        ctx.restore();

    function renderTrack() {
        drawAxes();
        const m = document.getElementById("MassValues").value;
        const pts = PreMSPathData[m];
        const val = document.getElementById("time-pre").value / 100;
        if (!pts || val <= 0) return;

        ctx.strokeStyle = "#00008B";
        ctx.lineWidth = 3;
        ctx.lineJoin = "round";
        ctx.beginPath();
        
        const start = pts[0];
        ctx.moveTo(TempConversion(start.T), YAxis(absMagnitudeFromLuminosity(start.L)));

        const res = 150;
        for (let j = 1; j <= res * val; j++) {
            const p = getSmoothPoint(pts, j / res);
            ctx.lineTo(TempConversion(p.T), YAxis(absMagnitudeFromLuminosity(p.L)));
        }
        ctx.stroke();

        const end = getSmoothPoint(pts, val);
        ctx.fillStyle = "#00008B";
        ctx.beginPath();
        ctx.arc(TempConversion(end.T), YAxis(absMagnitudeFromLuminosity(end.L)), 4, 0, Math.PI * 2);
        ctx.fill();
        ctx.strokeStyle = "white"; ctx.lineWidth = 1; ctx.stroke();
    }

    const slider = document.getElementById("time-pre");
    slider.oninput = function() {
        document.getElementById("pct-pre").innerText = this.value;
        renderTrack();
    };

    document.getElementById("MassValues").onchange = function() {
        slider.value = 0;
        document.getElementById("pct-pre").innerText = "0";
        renderTrack();
    };

    let playId = null;
    document.getElementById("play-pre").onclick = function() {
        if(playId) { clearInterval(playId); playId = null; this.innerText = "▶"; return; }
        this.innerText = "⏸";
        playId = setInterval(() => {
            let v = parseInt(slider.value);
            if(v >= 100) { clearInterval(playId); playId = null; this.innerText = "▶"; return; }
            slider.value = v + 1;
            document.getElementById("pct-pre").innerText = slider.value;
            renderTrack();
        }, 40);
    };

    window.onload = () => drawAxes();
